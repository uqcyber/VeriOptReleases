type_synonym ('a, 'b) Heap = "'a ⇒ 'b ⇒ Value"
type_synonym Free = "nat"
type_synonym ('a, 'b) DynamicHeap = "('a, 'b) Heap × Free"

fun h_load_field :: "'a ⇒ 'b ⇒ ('a, 'b) DynamicHeap ⇒ Value" where
  "h_load_field f r (h, n) = h f r"

fun h_store_field :: "'a ⇒ 'b ⇒ Value ⇒ ('a, 'b) DynamicHeap ⇒ ('a, 'b) DynamicHeap" where
  "h_store_field f r v (h, n) = (h(f := ((h f)(r := v))), n)"

fun h_new_inst :: "('a, 'b) DynamicHeap ⇒ ('a, 'b) DynamicHeap × Value" where
  "h_new_inst (h, n) = ((h,n+1), (ObjRef (Some n)))"

type_synonym FieldRefHeap = "(string, objref) DynamicHeap"

  "new_heap =  ((λf. λp. UndefVal), 0)"

subsection ‹Intraprocedural Semantics›

text ‹
Intraprocedural semantics are given as a small-step semantics.

Within the context of a graph, the configuration triple, (ID, MethodState, Heap),
is related to the subsequent configuration.
›

inductive step :: "IRGraph ⇒ (ID × MapState × FieldRefHeap) ⇒ (ID × MapState × FieldRefHeap) ⇒ bool"
  ("_ ⊢ _ → _" 55) for g where

  SequentialNode:
  "⟦is_sequential_node (kind g nid);
    nid' = (successors_of (kind g nid))!0⟧ 
    ⟹ g ⊢ (nid, m, h) → (nid', m, h)" |

  IfNode:
  "⟦kind g nid = (IfNode cond tb fb);
    g m ⊢ (kind g cond) ↦ val;
    nid' = (if val_to_bool val then tb else fb)⟧
    ⟹ g ⊢ (nid, m, h) → (nid', m, h)" |  

  EndNodes:
  "⟦is_AbstractEndNode (kind g nid);
    merge = any_usage g nid;
    is_AbstractMergeNode (kind g merge);

    i = find_index nid (inputs_of (kind g merge));
    phis = (phi_list g merge);
    inps = (phi_inputs g i phis);
    g m ⊢ inps ⟼ vs;

    m' = set_phis phis vs m⟧ 
    ⟹ g ⊢ (nid, m, h) → (merge, m', h)" |

  NewInstanceNode:
    "⟦kind g nid = (NewInstanceNode nid f obj nid');
      (h', ref) = h_new_inst h;
      m' = m_set nid ref m⟧ 
    ⟹ g ⊢ (nid, m, h) → (nid', m', h')" |

  LoadFieldNode:
    "⟦kind g nid = (LoadFieldNode nid f (Some obj) nid');
      g m ⊢ (kind g obj) ↦ ObjRef ref;
      h_load_field f ref h = v;
      m' = m_set nid v m⟧ 
    ⟹ g ⊢ (nid, m, h) → (nid', m', h)" |

  SignedDivNode:
    "⟦kind g nid = (SignedDivNode nid x y zero sb nxt);
      g m ⊢ (kind g x) ↦ v1;
      g m ⊢ (kind g y) ↦ v2;
      v = (intval_div v1 v2);
      m' = m_set nid v m⟧ 
    ⟹ g ⊢ (nid, m, h) → (nxt, m', h)" |

  SignedRemNode:
    "⟦kind g nid = (SignedRemNode nid x y zero sb nxt);
      g m ⊢ (kind g x) ↦ v1;
      g m ⊢ (kind g y) ↦ v2;
      v = (intval_mod v1 v2);
      m' = m_set nid v m⟧ 
    ⟹ g ⊢ (nid, m, h) → (nxt, m', h)" |

  StaticLoadFieldNode:
    "⟦kind g nid = (LoadFieldNode nid f None nid');
      h_load_field f None h = v;
      m' = m_set nid v m⟧ 
    ⟹ g ⊢ (nid, m, h) → (nid', m', h)" |

  StoreFieldNode:
    "⟦kind g nid = (StoreFieldNode nid f newval _ (Some obj) nid');
      g m ⊢ (kind g newval) ↦ val;
      g m ⊢ (kind g obj) ↦ ObjRef ref;
      h' = h_store_field f ref val h;
      m' = m_set nid val m⟧ 
    ⟹ g ⊢ (nid, m, h) → (nid', m', h')" |

  StaticStoreFieldNode:
    "⟦kind g nid = (StoreFieldNode nid f newval _ None nid');
      g m ⊢ (kind g newval) ↦ val;
      h' = h_store_field f None val h;
      m' = m_set nid val m⟧ 
    ⟹ g ⊢ (nid, m, h) → (nid', m', h')"

lemma IfNodeStepCases: 
  assumes "kind g nid = IfNode cond tb fb"
  assumes "g m ⊢ kind g cond ↦ v"
  assumes "g ⊢ (nid, m, h) → (nid', m, h)"
  shows "nid' ∈ {tb, fb}"
  using step.IfNode
  by (metis assms(1) assms(2) assms(3) insert_iff prod.inject stepDet)

lemma IfNodeSeq:
  shows "kind g nid = IfNode cond tb fb ⟶ ¬(is_sequential_node (kind g nid))"
  unfolding is_sequential_node.simps by simp

lemma IfNodeCond:
  assumes "kind g nid = IfNode cond tb fb"
  assumes "g ⊢ (nid, m, h) → (nid', m, h)"
  shows "∃ v. (g m ⊢ kind g cond ↦ v)"
  using assms(2,1) by (induct "(nid,m,h)" "(nid',m,h)" rule: step.induct; auto)

inductive step_top :: "Program ⇒ (IRGraph × ID × MapState) list × FieldRefHeap ⇒ (IRGraph × ID × MapState) list × FieldRefHeap ⇒ bool"
  ("_ ⊢ _ ⟶ _" 55) 
  for p where

  Lift:
  "⟦g ⊢ (nid, m, h) → (nid', m', h')⟧ 
    ⟹ p ⊢ ((g,nid,m)#stk, h) ⟶ ((g,nid',m')#stk, h')" |

  InvokeNodeStep:
  "⟦is_Invoke (kind g nid);

    callTarget = ir_callTarget (kind g nid);
    kind g callTarget = (MethodCallTargetNode targetMethod arguments);
    Some targetGraph = p targetMethod;

    g m ⊢ arguments ⟼ vs;
    m' = set_params m vs⟧
    ⟹ p ⊢ ((g,nid,m)#stk, h) ⟶ ((targetGraph,0,m')#(g,nid,m)#stk, h)" |

  ReturnNode:
  "⟦kind g nid = (ReturnNode (Some expr) _);
    g m ⊢ (kind g expr) ↦ v;

    c_m' = m_set c_nid v c_m;
    c_nid' = (successors_of (kind c_g c_nid))!0⟧ 
    ⟹ p ⊢ ((g,nid,m)#(c_g,c_nid,c_m)#stk, h) ⟶ ((c_g,c_nid',c_m')#stk, h)" |

  ReturnNodeVoid:
  "⟦kind g nid = (ReturnNode None _);
    c_m' = m_set c_nid (ObjRef (Some (2048))) c_m;
    
    c_nid' = (successors_of (kind c_g c_nid))!0⟧ 
    ⟹ p ⊢ ((g,nid,m)#(c_g,c_nid,c_m)#stk, h) ⟶ ((c_g,c_nid',c_m')#stk, h)" |

  UnwindNode:
  "⟦kind g nid = (UnwindNode exception);

    g m ⊢ (kind g exception) ↦ e;
     
    kind c_g c_nid = (InvokeWithExceptionNode _ _ _ _ _ _ exEdge);

    c_m' = m_set c_nid e c_m⟧
  ⟹ p ⊢ ((g,nid,m)#(c_g,c_nid,c_m)#stk, h) ⟶ ((c_g,exEdge,c_m')#stk, h)"
