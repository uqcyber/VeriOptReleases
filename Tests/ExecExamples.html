(* NB: The starting state is duplicated causing the program to be executed twice
       The reason for this is that the top step of ReturnNode empties
       the state list to signal completion, this means we can't access the state

   Discuss ways to fix this
 *)

fun id_and_state :: "IRGraph × ID × MapState × Value list ⇒ ID × MapState" where
  "id_and_state (g,i,m,ps) = (i,m)"

inductive exec_graph :: "IRGraph ⇒ Value list ⇒ (ID × MapState) ⇒ Trace ⇒ bool" ("_|_↝_|_")
  where
  "⟦s0 = (g, 0, new_map_state, ps);
    (λx. Some g) ⊢ ([s0, s0], new_heap) | [] ⟶* ((end # xs), heap) | l⟧
    ⟹ exec_graph g ps (id_and_state end) l"
code_pred (modes: i ⇒ i ⇒ o * o ⇒ o ⇒ bool as execE) "exec_graph" .
