inductive static_test :: "IRGraph ⇒ Value list ⇒ Value ⇒ bool"
  where
  "⟦state = new_map ps;
    (λx. Some g) ⊢ ([(g, 0, state), (g, 0, state)], new_heap) | [] ⟶* ((end # xs), heap) | l ⟧
    ⟹ static_test g ps (m_val (prod.snd (prod.snd end)) 0)"

code_pred (modes: i ⇒ i ⇒ i ⇒ bool as testE) static_test .

(* program_test runs a static initialisation block first (to initialise static fields etc.),
   then a named method.
*)
inductive program_test :: "Program ⇒ Signature ⇒ Value list ⇒ Value ⇒ bool"
  where
  InitStatics:
  "⟦Some init = prog '''';
    state0 = new_map [];
    prog ⊢ ([(init, 0, state0), (init, 0, state0)], new_heap) | [] ⟶* ((end1 # xs1), heap1) | l1;
    
    Some g = prog m;
    state1 = new_map ps;
    prog ⊢ ([(g, 0, state1), (g, 0, state1)], heap1) | [] ⟶* ((end2 # xs2), heap2) | l2 ⟧
    ⟹ program_test prog m ps (m_val (prod.snd (prod.snd end2)) 0)" |

  NoStatics:
  "⟦'''' ∉ dom prog;
    Some g = prog m;
    state = new_map ps;
    prog ⊢ ([(g, 0, state), (g, 0, state)], new_heap) | [] ⟶* ((end2 # xs2), heap2) | l2 ⟧
    ⟹ program_test prog m ps (m_val (prod.snd (prod.snd end2)) 0)"

code_pred (modes: i ⇒ i ⇒ i ⇒ i ⇒ bool as testP) program_test .
